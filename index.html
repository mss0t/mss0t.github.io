def make_truth_table(bit_names, logic):
  print("|", end='')
  for x in bit_names:
    print(x, end = '|')
  print("OUTPUT|")
  print("|-")
  for i in range(2**len(bit_names)):
    n = i
    bool_array = []
    for j in range(len(bit_names)):
      bool_array = [(n%2 == 1)] + bool_array
      n = n//2
    output = logic(*bool_array)
    print("|", end='')
    for x in bool_array:
      print(int(x), end = '|')
    print(str(int(output)) + "|")
